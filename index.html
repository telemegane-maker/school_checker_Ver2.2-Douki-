/* 修正した importData 関数 */ function importData() { const input = document.getElementById('syncData').value.trim(); if (!input) return; try { const receivedData = JSON.parse(atob(input)); // 【修正ポイント：上書きではなく「合算（マージ）」する】 // 届いたデータのすべてのキー（班のID）をループで確認 for (let id in receivedData) { const currentVal = state.peerData[id] || 0; const newVal = receivedData[id] || 0; // 手元にある相方のデータと、今届いたデータを比較して、 // より進んでいる方（大きい数値）を保持する if (newVal > currentVal) { state.peerData[id] = newVal; } } renderGrid(); save(); alert("全員の進捗を合体させました！"); document.getElementById('syncData').value = ""; } catch (e) { alert("データの形式が正しくありません。"); } }
